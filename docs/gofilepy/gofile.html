 65    def _download_file_from_direct_link(self, direct_link, out_dir="./"):
 66        fn = direct_link.rsplit('/', 1)[1]
 67        resp = requests.get(direct_link, stream=True, allow_redirects=None)
 68        if resp.status_code != 200:
 69            raise GofileApiException("Could not download file", code=resp.status_code)
 70
 71        out_path = os.path.join(out_dir, fn)
 72        with open(out_path, 'wb') as f:
 73            for chunk in resp.iter_content(chunk_size=1024):
 74                if chunk:
 75                    f.write(chunk)
 76        return out_path

333    def reload (self):
334        """Reloads any new updates to content.  If is_unknown_type = True then reload() returns updated content"""
335        if self.is_folder_type or (self.is_unknown_type and self.parent_id == None):
336            resp, data = self._client._get_content_raw_resp(self.content_id)
337
338            if self.is_unknown_type:
339                content = GofileContent.__init_from_resp__(resp, client=self._client)
340
341            elif self.is_folder_type:
342                self._override_from_dict(data)
343                return self
344        
345        elif (self.is_unknown_type or self.is_file_type) and self.parent_id:
346            resp, data = self._client._get_content_raw_resp(self.parent_id)
347            content_data = data["contents"].get(self.content_id, None)
348            content = None
349            if content_data:
350                content = GofileContent.__init_from_resp__({"data": content_data}, client=self._client)
351
352                if self.is_file_type:
353                    self._override_from_dict(content_data)
354
355            return content
356
357        else:
358            raise NotImplemented

371        elif _type == "folder":
